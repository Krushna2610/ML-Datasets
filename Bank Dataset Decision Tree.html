def my_model(model):
    model.fit(X_train,y_train)
    y_pred_train=model.predict(X_train)
    y_pred_test=model.predict(X_test)
    print("Train Data")
    print(classification_report(y_train,y_pred_train))
    print("Test Data")
    print(classification_report(y_test,y_pred_test))
    return model
